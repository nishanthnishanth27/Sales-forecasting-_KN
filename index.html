# \u2500\u2500 Step 1: Load & Explore \u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500 import pandas as pd import matplotlib.pyplot as plt df = pd.read_csv('sales_data.csv', parse_dates=['date']) df.set_index('date', inplace=True) print(df.head()) # First 5 rows print(df.info()) # Column types & nulls print(df.describe()) # Statistical summary df['sales'].plot(figsize=(14,4), title='Daily Sales') plt.show()
02
Handle Missing Values & Outliers
Interpolate \u00b7 Z-Score \u00b7 IQR Capping
BEFORE \u2014 RAW WITH OUTLIERS & NULLS
AFTER \u2014 CLEANED & INTERPOLATED
# \u2500\u2500 Step 2: Clean Data \u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500 from scipy import stats import numpy as np # Fill missing with linear interpolation df['sales'] = df['sales'].interpolate(method='linear') # Z-score outlier detection z = np.abs(stats.zscore(df['sales'])) print(f"Outliers (z>3): {(z>3).sum()}") # Cap with IQR Q1 = df['sales'].quantile(0.25) Q3 = df['sales'].quantile(0.75) IQR = Q3 - Q1 df['sales'] = df['sales'].clip(Q1 - 1.5*IQR, Q3 + 1.5*IQR)
03
Create Time-Based Features
Lag Features \u00b7 Rolling Averages \u00b7 Calendar
LAG-7 vs ACTUAL SALES
ROLLING 7-DAY & 30-DAY MOVING AVERAGE
# \u2500\u2500 Step 3: Feature Engineering \u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500 df['day_of_week'] = df.index.dayofweek # 0=Mon, 6=Sun df['month'] = df.index.month df['quarter'] = df.index.quarter df['year'] = df.index.year # Lag features (previous sales) df['lag_7'] = df['sales'].shift(7) df['lag_14'] = df['sales'].shift(14) df['lag_30'] = df['sales'].shift(30) # Rolling statistics df['roll_7'] = df['sales'].rolling(7).mean() df['roll_30'] = df['sales'].rolling(30).mean() df['roll_std'] = df['sales'].rolling(7).std() df.dropna(inplace=True)
04
Split Data \u2014 Train & Test Sets
Chronological Split \u00b7 No Data Leakage
TRAIN (2020\u20132023) VS TEST (2024) \u2014 CHRONOLOGICAL SPLIT
# \u2500\u2500 Step 4: Train/Test Split \u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500\u2500 split_date = '2024-01-01' train = df[df.index < split_date] test = df[df.index >= split_date] features = ['lag_7', 'lag_14', 'lag_30', 'roll_7', 'roll_30', 'roll_std', 'day_of_week', 'month', 'quarter'] X_train, y_train = train[features], train['sales'] X_test, y_test = test[features], test['sales'] print(f"Train rows: {len(train)} | Test rows: {len(test)}")
