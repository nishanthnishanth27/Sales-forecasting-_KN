# Step 1: Load & Explore import pandas as pd import matplotlib.pyplot as plt df = pd.read_csv('sales_data.csv', parse_dates=['date']) df.set_index('date', inplace=True) print(df.head()) print(df.info()) print(df.describe()) df['sales'].plot(figsize=(14,4), title='Daily Sales') plt.show()
02
Handle Missing Values & Outliers
Interpolate · Z-Score · IQR Capping
BEFORE — RAW WITH OUTLIERS & NULLS
AFTER — CLEANED & INTERPOLATED
# Step 2: Clean Data from scipy import stats import numpy as np df['sales'] = df['sales'].interpolate(method='linear') z = np.abs(stats.zscore(df['sales'])) print(f"Outliers (z>3): {(z>3).sum()}") Q1 = df['sales'].quantile(0.25) Q3 = df['sales'].quantile(0.75) IQR = Q3 - Q1 df['sales'] = df['sales'].clip(Q1 - 1.5*IQR, Q3 + 1.5*IQR)
03
Create Time-Based Features
Lag Features · Rolling Averages · Calendar
LAG-7 VS ACTUAL SALES
7-DAY & 30-DAY MOVING AVERAGE
# Step 3: Feature Engineering df['day_of_week'] = df.index.dayofweek df['month'] = df.index.month df['quarter'] = df.index.quarter df['year'] = df.index.year df['lag_7'] = df['sales'].shift(7) df['lag_14'] = df['sales'].shift(14) df['lag_30'] = df['sales'].shift(30) df['roll_7'] = df['sales'].rolling(7).mean() df['roll_30'] = df['sales'].rolling(30).mean() df['roll_std'] = df['sales'].rolling(7).std() df.dropna(inplace=True)
04
Split Data — Train & Test Sets
Chronological Split · No Data Leakage
TRAIN (2020–2023) VS TEST (2024) — CHRONOLOGICAL SPLIT
# Step 4: Chronological Train/Test Split split_date = '2024-01-01' train = df[df.index < split_date] test = df[df.index >= split_date] features = ['lag_7','lag_14','lag_30','roll_7','roll_30','roll_std','day_of_week','month','quarter'] X_train, y_train = train[features], train['sales'] X_test, y_test = test[features], test['sales'] print(f"Train: {len(train)} | Test: {len(test)}")
05
Train Forecasting Models
ARIMA · Prophet · XGBoost
TRAINING LOSS CONVERGENCE — ALL 3 MODELS
# Step 5: Train Models from statsmodels.tsa.arima.model import ARIMA arima = ARIMA(y_train, order=(5,1,0)).fit() arima_pred = arima.forecast(steps=len(y_test)) from prophet import Prophet pdf = train.reset_index().rename(columns={'date':'ds','sales':'y'}) m = Prophet(seasonality_mode='multiplicative') m.fit(pdf) future = m.make_future_dataframe(periods=len(test)) prophet_pred = m.predict(future).tail(len(test))['yhat'].values from xgboost import XGBRegressor xgb = XGBRegressor(n_estimators=300, max_depth=6, learning_rate=0.05) xgb.fit(X_train, y_train) xgb_pred = xgb.predict(X_test)
06
Evaluate Model Performance
MAE · RMSE · MAPE · Model Comparison
RMSE COMPARISON — ALL MODELS
RESIDUALS DISTRIBUTION — XGBOOST
# Step 6: Evaluate from sklearn.metrics import mean_absolute_error, mean_squared_error import numpy as np def mape(y_true, y_pred): return np.mean(np.abs((y_true - y_pred) / y_true)) * 100 for name, pred in {'XGBoost':xgb_pred,'ARIMA':arima_pred,'Prophet':prophet_pred}.items(): mae = mean_absolute_error(y_test, pred) rmse = np.sqrt(mean_squared_error(y_test, pred)) mp = mape(y_test.values, pred) print(f"{name}: MAE={mae:.1f} RMSE={rmse:.1f} MAPE={mp:.2f}%")
