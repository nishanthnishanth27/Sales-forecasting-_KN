# Step 1: Load & Explore import pandas as pd import matplotlib.pyplot as plt df = pd.read_csv('sales_data.csv', parse_dates=['date']) df.set_index('date', inplace=True) # Basic EDA print(df.head()) # First 5 rows print(df.info()) # Dtypes, nulls print(df.describe()) # Stats summary # Plot raw time series df['sales'].plot(title='Raw Sales', figsize=(14,4)) plt.show()

BEFORE CLEANING — MISSING & OUTLIERS
AFTER CLEANING — INTERPOLATED & CAPPED
# Step 2: Clean Data from scipy import stats import numpy as np # Fill missing with linear interpolation df['sales'] = df['sales'].interpolate(method='linear') # Detect outliers using Z-score z_scores = np.abs(stats.zscore(df['sales'])) outliers = df[z_scores > 3] print(f"Outliers found: {len(outliers)}") # Cap using IQR method Q1 = df['sales'].quantile(0.25) Q3 = df['sales'].quantile(0.75) IQR = Q3 - Q1 df['sales'] = df['sales'].clip(Q1 - 1.5*IQR, Q3 + 1.5*IQR)
03
Create Time-Based Features
Lag Features · Rolling Averages · Calendar Features
LAG FEATURES CORRELATION
ROLLING AVERAGE (7-DAY vs 30-DAY)
# Step 3: Feature Engineering df['day_of_week'] = df.index.dayofweek df['month'] = df.index.month df['quarter'] = df.index.quarter df['year'] = df.index.year df['week'] = df.index.isocalendar().week # Lag features df['lag_7'] = df['sales'].shift(7) df['lag_14'] = df['sales'].shift(14) df['lag_30'] = df['sales'].shift(30) # Rolling statistics df['roll_7_mean'] = df['sales'].rolling(7).mean() df['roll_30_mean'] = df['sales'].rolling(30).mean() df['roll_7_std'] = df['sales'].rolling(7).std() df.dropna(inplace=True)
04
Split Data — Train & Test Sets
Chronological Split · No Data Leakage
TRAIN / TEST SPLIT — CHRONOLOGICAL (2020–2023 TRAIN · 2024 TEST)
# Step 4: Chronological Train/Test Split split_date = '2024-01-01' train = df[df.index < split_date] test = df[df.index >= split_date] features = ['lag_7', 'lag_14', 'lag_30', 'roll_7_mean', 'roll_30_mean', 'roll_7_std', 'day_of_week', 'month', 'quarter'] X_train, y_train = train[features], train['sales'] X_test, y_test = test[features], test['sales'] print(f"Train: {len(train)} rows | Test: {len(test)} rows")
